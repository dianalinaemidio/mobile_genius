CREATE TABLE CELULAR(
ID INT PRIMARY KEY AUTO_INCREMENT,
marca VARCHAR(255), 
modelo VARCHAR(255), 
preco VARCHAR(255), 
quantidade VARCHAR(255), 
descricao VARCHAR(255),
image VARCHAR (255);

);


CREATE TABLE USERS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    USERNAME VARCHAR(255),
    PASSWORD VARCHAR(255),
    TYPE BOOLEAN
);
